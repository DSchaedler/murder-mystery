def initialize(options = {})
  @state = options[:state]
  @x = options[:x]
  @y = options[:y]
  @w = options[:w]

  @sprite = :nil

  @allowable_states = %i[empty checked_true checked_false]
end

def tick
  args = $gtk.args
  if args.inputs.mouse.up && args.geometry.inside_rect?([args.inputs.mouse.x, args.inputs.mouse.y, 1, 1], [@x, @y, @w, @h])
    case @state
    when :empty || :checked_false
      @state = :checked_true
    when :checked_true
      @state = :empty
    end
  end

  if $app.rmb.up && args.geometry.inside_rect?([args.inputs.mouse.x, args.inputs.mouse.y, 1, 1], [@x, @y, @w, @h])
    case @state
    when :empty || :checked_true
      @state = :checked_false
    when :checked_false
      @state = :empty
    end
  end

  case @state
  when :checked_false
    @sprite = FALSE_MARK
  when :checked_true
    @sprite = TRUE_MARK
  when :empty
    @sprite = :nil
  end
end
